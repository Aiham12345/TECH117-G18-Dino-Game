Introduction

This project uses an Arduino Uno, an HC-SR04 ultrasonic sensor, a passive buzzer, and red/yellow/green LEDs to provide proximity-based alerts, similar to automotive collision detection systems.

Circuit Diagram

The ultrasonic sensor measures distance; the Arduino activates LEDs and the buzzer according to distance thresholds.

Environmental Impact Assessment

  • Materials: The Arduino Uno is RoHS and WEEE compliant, limiting hazardous substances.
  • Energy Efficiency: The circuit draws minimal power (~50 mA at 5V), ideal for USB or solar power.
  • Reusability: All components are modular and reusable, reducing electronic waste.
  • Packaging & Shipping: Ordering all components from a single supplier helps minimize carbon footprint.

Arduino Code

The following Arduino code controls the system, lighting LEDs and activating the buzzer based on distance readings from the HC-SR04 sensor.

  
  
#include 

LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

#define PIN_BUTTON 7   // pushbutton pin, with internal pull-up
#define LCD_DINO_CHAR_INDEX 0

byte dinoChar[8] = {
  B00100,
  B00100,
  B01110,
  B10100,
  B00100,
  B01100,
  B01010,
  B01001
};

bool gameRunning = false;
bool jumping = false;
int jumpFrames = 0;
const int JUMP_DURATION = 4;

int obstacleX = 15;
unsigned long score = 0;

void setup() {
  pinMode(PIN_BUTTON, INPUT_PULLUP);
  lcd.begin(16, 2);
  lcd.createChar(LCD_DINO_CHAR_INDEX, dinoChar);
  showMenu();
}

void showMenu() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("DINO GAME");
  lcd.setCursor(0, 1);
  lcd.print("Press button");
}

void startGame() {
  gameRunning = true;
  jumping = false;
  jumpFrames = 0;
  obstacleX = 15;
  score = 0;
}

void drawGame() {
  lcd.clear();

  // Draw obstacle
  if (obstacleX >= 0 && obstacleX < 16) {
    lcd.setCursor(obstacleX, 1);
    lcd.print("|");
  }

  // Draw dino
  int dinoRow = jumping ? 0 : 1;
  lcd.setCursor(1, dinoRow);
  lcd.write(byte(LCD_DINO_CHAR_INDEX));

  // Draw score
  lcd.setCursor(12, 0);
  lcd.print(score);
}

void loop() {
  int btn = digitalRead(PIN_BUTTON);

  if (!gameRunning) {
    // menu state
    if (btn == LOW) {
      // button pressed → start game
      delay(50);  // basic debounce
      if (digitalRead(PIN_BUTTON) == LOW) {
        startGame();
      }
    }
    return;
  }

  // in‑game logic
  if (btn == LOW && !jumping) {
    // jump
    jumping = true;
    jumpFrames = 0;
    delay(50);  // debounce
  }

  if (jumping) {
    jumpFrames++;
    if (jumpFrames >= JUMP_DURATION) {
      jumping = false;
    }
  }

  obstacleX--;
  if (obstacleX < -1) {
    obstacleX = 15;
    score++;
  }

  // collision detection
  if (!jumping && obstacleX == 1) {
    // hit obstacle
    lcd.clear();
    lcd.setCursor(3, 0);
    lcd.print("GAME OVER");
    lcd.setCursor(3, 1);
    lcd.print("Score:");
    lcd.print(score);
    delay(2000);
    gameRunning = false;
    showMenu();
    return;
  }

  drawGame();
  delay(200);
}

Conclusion

The system effectively demonstrates distance-based alerting using Arduino. It’s affordable, educational, and sustainable through reusable components.
